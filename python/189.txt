class Solution(object):
    def rotate(self, nums, k):
        """
        :type nums: List[int]
        :type k: int
        :rtype: None Do not return anything, modify nums in-place instead.
        """
        k = k % len(nums)
        nums1 = nums[0: len(nums)-k]
        nums2 = nums[(len(nums)-k):]
        num = nums2 + nums1
        for i in range(len(num)):
            nums[i] = num[i]
        return nums
